import re

def assess_password_strength(password):
    strength_criteria = {
        "length": len(password) >= 8,
        "uppercase": bool(re.search(r'[A-Z]', password)),
        "lowercase": bool(re.search(r'[a-z]', password)),
        "digit": bool(re.search(r'[0-9]', password)),
        "special": bool(re.search(r'[@$!%*?&]', password)),
    }
    strength_score = sum(strength_criteria.values())
    if strength_score == 5:
        return "Strong password"
    elif strength_score == 4:
        return "Moderately strong password"
    elif strength_score == 3:
        return "Weak password"
    else:
        return "Very weak password"
def main():
    while True: 
        print("\nPassword Strength Assessment Tool")
        password = input("Enter your password: ")
        strength = assess_password_strength(password)
        print(f"Password strength: {strength}")
        if strength == "Very weak password":
            print("Suggestions:")
            print("- Use at least 8 characters.")
            print("- Include both uppercase and lowercase letters.")
            print("- Add at least one number.")
            print("- Include special characters like @$!%*?&.")
        elif strength == "Weak password":
            print("Suggestions:")
            print("- Consider increasing the length.")
            print("- Mix uppercase and lowercase letters.")
            print("- Add numbers and special characters.")
        elif strength == "Moderately strong password":
            print("Suggestions:")
            print("- Consider adding more special characters or increasing length for better security.")
        choice = input("\nDo you want to enter another password? (yes/no): ").strip().lower()
        if choice != 'yes':
            print("Exiting the tool. Goodbye!")
            break
if __name__ == "__main__":
    main()
